115  template <class T, class A>
116  xcsv_tensor<T, A> load_csv(std::istream& stream,
117  const char delimiter,
118  const std::size_t skip_rows,
119  const std::ptrdiff_t max_rows,
120  const std::string comments)
121  {
122  using tensor_type = xcsv_tensor<T, A>;
123  using storage_type = typename tensor_type::storage_type;
124  using size_type = typename tensor_type::size_type;
125  using inner_shape_type = typename tensor_type::inner_shape_type;
126  using inner_strides_type = typename tensor_type::inner_strides_type;
127  using output_iterator = std::back_insert_iterator<storage_type>;
128 
129  storage_type data;
130  size_type nbrow = 0, nbcol = 0, nhead = 0;
131  {
132  output_iterator output(data);
133  std::string row, cell;
134  while (std::getline(stream, row))
135  {
136  if (nhead < skip_rows)
137  {
138  ++nhead;
139  continue;
140  }
141  if (std::equal(comments.begin(), comments.end(), row.begin()))
142  {
143  continue;
144  }
145  if (0 < max_rows && max_rows <= static_cast<const long long>(nbrow))
146  {
147  break;
148  }
149  std::stringstream row_stream(row);
150  nbcol = detail::load_csv_row<size_type, T, output_iterator>(row_stream, output, cell, delimiter);
151  ++nbrow;
152  }
153  }
154  inner_shape_type shape = {nbrow, nbcol};
155  inner_strides_type strides; // no need for initializer list for stack-allocated strides_type
156  size_type data_size = compute_strides(shape, layout_type::row_major, strides);
157  // Sanity check for data size.
158  if (data.size() != data_size)
159  {
160  XTENSOR_THROW(std::runtime_error, "Inconsistent row lengths in CSV");
161  }
162  return tensor_type(std::move(data), std::move(shape), std::move(strides));
163  }

171  template <class E>
172  void dump_csv(std::ostream& stream, const xexpression<E>& e)
173  {
174  using size_type = typename E::size_type;
175  const E& ex = e.derived_cast();
176  if (ex.dimension() != 2)
177  {
178  XTENSOR_THROW(std::runtime_error, "Only 2-D expressions can be serialized to CSV");
179  }
180  size_type nbrows = ex.shape()[0], nbcols = ex.shape()[1];
181  auto st = ex.stepper_begin(ex.shape());
182  for (size_type r = 0; r != nbrows; ++r)
183  {
184  for (size_type c = 0; c != nbcols; ++c)
185  {
186  stream << *st;
187  if (c != nbcols - 1)
188  {
189  st.step(1);
190  stream << ',';
191  }
192  else
193  {
194  st.reset(1);
195  st.step(0);
196  stream << std::endl;
197  }
198  }
199  }
