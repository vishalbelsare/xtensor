368  template <class D>
369  template <class E>
370  inline auto xchunked_semantic<D>::assign_xexpression(const xexpression<E>& e) -> derived_type&
371  {
372  using shape_type = std::decay_t<decltype(this->derived_cast().shape())>;
373  using size_type = typename shape_type::size_type;
374  const auto& chunk_shape = this->derived_cast().chunk_shape();
375  auto& chunks = this->derived_cast().chunks();
376  size_t dimension = this->derived_cast().dimension();
377  xstrided_slice_vector sv(chunk_shape.size()); // element slice corresponding to chunk
378  std::transform(chunk_shape.begin(), chunk_shape.end(), sv.begin(),
379  [](auto size) { return range(0, size); });
380  shape_type ic(dimension); // index of chunk, initialized to 0...
381  size_type ci = 0;
382  for (auto& chunk: chunks)
383  {
384  auto rhs = strided_view(e.derived_cast(), sv);
385  auto rhs_shape = rhs.shape();
386  if (rhs_shape != chunk_shape)
387  {
388  xstrided_slice_vector esv(chunk_shape.size()); // element slice in edge chunk
389  std::transform(rhs_shape.begin(), rhs_shape.end(), esv.begin(),
390  [](auto size) { return range(0, size); });
391  noalias(strided_view(chunk, esv)) = rhs;
392  }
393  else
394  {
395  noalias(chunk) = rhs;
396  }
397  bool last_chunk = ci == chunks.size() - 1;
398  if (!last_chunk)
399  {
400  size_type di = dimension - 1;
401  while (true)
402  {
403  if (ic[di] + 1 == chunks.shape()[di])
404  {
405  ic[di] = 0;
406  sv[di] = range(0, chunk_shape[di]);
407  if (di == 0)
408  {
409  break;
410  }
411  else
412  {
413  di--;
414  }
415  }
416  else
417  {
418  ic[di] += 1;
419  sv[di] = range(ic[di] * chunk_shape[di], (ic[di] + 1) * chunk_shape[di]);
420  break;
421  }
422  }
423  }
424  ++ci;
425  }
426  return this->derived_cast();
427  }

627  inline void xchunked_array<CS>::resize(S1&& shape, S2&& chunk_shape, layout_type chunk_memory_layout)
628  {
629  // compute chunk number in each dimension (shape_of_chunks)
630  std::vector<std::size_t> shape_of_chunks(shape.size());
631  std::transform
632  (
633  shape.cbegin(), shape.cend(),
634  chunk_shape.cbegin(),
635  shape_of_chunks.begin(),
636  [](auto s, auto cs)
637  {
638  std::size_t cn = s / cs;
639  if (s % cs > 0)
640  cn += std::size_t(1); // edge_chunk
641  return cn;
642  }
643  );
644 
645  detail::chunk_helper<CS>::resize(m_chunks, shape_of_chunks, chunk_shape, chunk_memory_layout);
646 
647  m_shape = xtl::forward_sequence<shape_type, S1>(shape);
648  m_chunk_shape = xtl::forward_sequence<shape_type, S2>(chunk_shape);
649  }
