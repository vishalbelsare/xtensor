153  template <class F, class E>
154  inline auto accumulator_init_with_f(F&& f, E& e, std::size_t axis)
155  {
156  // this function is the equivalent (but hopefully faster) to (if axis == 1)
157  // e[:, 0, :, :, ...] = f(e[:, 0, :, :, ...])
158  // so that all "first" values are initialized in a first pass
159 
160  std::size_t outer_loop_size, inner_loop_size, pos = 0;
161  std::size_t outer_stride, inner_stride;
162 
163  auto set_loop_sizes = [&outer_loop_size, &inner_loop_size](auto first, auto last, std::ptrdiff_t ax) {
164  outer_loop_size = std::accumulate(first, first + ax,
165  std::size_t(1), std::multiplies<std::size_t>());
166  inner_loop_size = std::accumulate(first + ax + 1, last,
167  std::size_t(1), std::multiplies<std::size_t>());
168  };
169 
170  // Note: add check that strides > 0
171  auto set_loop_strides = [&outer_stride, &inner_stride](auto first, auto last, std::ptrdiff_t ax) {
172  outer_stride = static_cast<std::size_t>(ax == 0 ? 1 : *std::min_element(first, first + ax));
173  inner_stride = static_cast<std::size_t>((ax == std::distance(first, last) - 1) ? 1 : *std::min_element(first + ax + 1, last));
174  };
175 
176  set_loop_sizes(e.shape().begin(), e.shape().end(), static_cast<std::ptrdiff_t>(axis));
177  set_loop_strides(e.strides().begin(), e.strides().end(), static_cast<std::ptrdiff_t>(axis));
178 
179  if (e.layout() == layout_type::column_major)
180  {
181  // swap for better memory locality (smaller stride in the inner loop)
182  std::swap(outer_loop_size, inner_loop_size);
183  std::swap(outer_stride, inner_stride);
184  }
185 
186  for (std::size_t i = 0; i < outer_loop_size; ++i)
187  {
188  pos = i * outer_stride;
189  for (std::size_t j = 0; j < inner_loop_size; ++j)
190  {
191  e.storage()[pos] = f(e.storage()[pos]);
192  pos += inner_stride;
193  }
194  }
195  }

197  template <class F, class E>
198  inline auto accumulator_impl(F&& f, E&& e, std::size_t axis, evaluation_strategy::immediate_type)
199  {
200  using init_type = typename F::init_value_type;
201  using accumulate_functor_type = typename F::accumulate_functor_type;
202  using expr_value_type = typename std::decay_t<E>::value_type;
203  //using return_type = std::conditional_t<std::is_same<init_type, void>::value, typename std::decay_t<E>::value_type, init_type>;
204 
205  using return_type = std::decay_t<decltype(std::declval<accumulate_functor_type>()(std::declval<init_type>(),
206  std::declval<expr_value_type>()))>;
207  using result_type = xaccumulator_return_type_t<std::decay_t<E>, return_type>;
208 
209  if (axis >= e.dimension())
210  {
211  XTENSOR_THROW(std::runtime_error, "Axis larger than expression dimension in accumulator.");
212  }
213 
214  result_type result = e; // assign + make a copy, we need it anyways
215 
216  if(result.shape(axis) != std::size_t(0))
217  {
218  std::size_t inner_stride = static_cast<std::size_t>(result.strides()[axis]);
219  std::size_t outer_stride = 1; // this is either going row- or column-wise (strides.back / strides.front)
220  std::size_t outer_loop_size = 0;
221  std::size_t inner_loop_size = 0;
222  std::size_t init_size = e.shape()[axis] != std::size_t(1) ? std::size_t(1) : std::size_t(0);
223 
224  auto set_loop_sizes = [&outer_loop_size, &inner_loop_size, init_size](auto first, auto last, std::ptrdiff_t ax) {
225  outer_loop_size = std::accumulate(first,
226  first + ax,
227  init_size, std::multiplies<std::size_t>());
228 
229  inner_loop_size = std::accumulate(first + ax,
230  last,
231  std::size_t(1), std::multiplies<std::size_t>());
232  };
233 
234  if (result_type::static_layout == layout_type::row_major)
235  {
236  set_loop_sizes(result.shape().cbegin(), result.shape().cend(), static_cast<std::ptrdiff_t>(axis));
237  }
238  else
239  {
240  set_loop_sizes(result.shape().cbegin(), result.shape().cend(), static_cast<std::ptrdiff_t>(axis + 1));
241  std::swap(inner_loop_size, outer_loop_size);
242  }
243 
244  std::size_t pos = 0;
245 
246  inner_loop_size = inner_loop_size - inner_stride;
247 
248  // activate the init loop if we have an init function other than identity
249  if (!std::is_same<std::decay_t<typename F::init_functor_type>,
250  typename detail::accumulator_identity<init_type>>::value)
251  {
252  accumulator_init_with_f(xt::get<1>(f), result, axis);
253  }
254 
255  pos = 0;
256  for (std::size_t i = 0; i < outer_loop_size; ++i)
257  {
258  for (std::size_t j = 0; j < inner_loop_size; ++j)
259  {
260  result.storage()[pos + inner_stride] = xt::get<0>(f)(result.storage()[pos],
261  result.storage()[pos + inner_stride]);
262  pos += outer_stride;
263  }
264  pos += inner_stride;
265  }
266  }
267  return result;
268  }

270  template <class F, class E>
271  inline auto accumulator_impl(F&& f, E&& e, evaluation_strategy::immediate_type)
272  {
273  using init_type = typename F::init_value_type;
274  using expr_value_type = typename std::decay_t<E>::value_type;
275  using accumulate_functor_type = typename F::accumulate_functor_type;
276  using return_type = std::decay_t<decltype(std::declval<accumulate_functor_type>()(std::declval<init_type>(),
277  std::declval<expr_value_type>()))>;
278  //using return_type = std::conditional_t<std::is_same<init_type, void>::value, typename std::decay_t<E>::value_type, init_type>;
279  using result_type = xaccumulator_return_type_t<std::decay_t<E>, return_type>;
280 
281  std::size_t sz = e.size();
282  auto result = result_type::from_shape({sz});
283 
284  if(sz != std::size_t(0))
285  {
286  auto it = e.template begin<XTENSOR_DEFAULT_TRAVERSAL>();
287  result.storage()[0] = xt::get<1>(f)(*it);
288  ++it;
289 
290  for (std::size_t idx = 0; it != e.template end<XTENSOR_DEFAULT_TRAVERSAL>(); ++it)
291  {
292  result.storage()[idx + 1] = xt::get<0>(f)(result.storage()[idx], *it);
293  ++idx;
294  }
295  }
296  return result;
297  }
298  }
