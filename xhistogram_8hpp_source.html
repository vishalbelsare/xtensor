49  template <class R = double, class E1, class E2, class E3>
50  inline auto histogram_imp(E1&& data, E2&& bin_edges, E3&& weights, bool density, bool equal_bins)
51  {
52  using size_type = common_size_type_t<std::decay_t<E1>, std::decay_t<E2>, std::decay_t<E3>>;
53  using value_type = typename std::decay_t<E3>::value_type;
54 
55  XTENSOR_ASSERT(data.dimension() == 1);
56  XTENSOR_ASSERT(weights.dimension() == 1);
57  XTENSOR_ASSERT(bin_edges.dimension() == 1);
58  XTENSOR_ASSERT(weights.size() == data.size());
59  XTENSOR_ASSERT(bin_edges.size() >= 2);
60  XTENSOR_ASSERT(std::is_sorted(bin_edges.cbegin(), bin_edges.cend()));
61 
62  size_t n_bins = bin_edges.size() - 1;
63  xt::xtensor<value_type, 1> count = xt::zeros<value_type>({ n_bins });
64 
65  if (equal_bins)
66  {
67  std::array<typename std::decay_t<E2>::value_type, 2> bounds = xt::minmax(bin_edges)();
68  auto left = static_cast<double>(bounds[0]);
69  auto right = static_cast<double>(bounds[1]);
70  double norm = 1. / (right - left);
71  for (size_t i = 0; i < data.size(); ++i)
72  {
73  auto v = static_cast<double>(data(i));
74  // left and right are not bounds of data
75  if ( v >= left & v < right )
76  {
77  auto i_bin = static_cast<size_t>(static_cast<double>(n_bins) * (v - left) * norm);
78  count(i_bin) += weights(i);
79  }
80  else if ( v == right )
81  {
82  count(n_bins - 1) += weights(i);
83  }
84  }
85  }
86  else
87  {
88  auto sorter = xt::argsort(data);
89 
90  size_type ibin = 0;
91 
92  for (auto& idx : sorter)
93  {
94  auto item = data[idx];
95 
96  if (item < bin_edges[0])
97  {
98  continue;
99  }
100 
101  if (item > bin_edges[n_bins])
102  {
103  break;
104  }
105 
106  while (item >= bin_edges[ibin + 1] && ibin < n_bins - 1)
107  {
108  ++ibin;
109  }
110 
111  count[ibin] += weights[idx];
112  }
113  }
114 
115  xt::xtensor<R, 1> prob = xt::cast<R>(count);
116 
117  if (density)
118  {
119  R n = static_cast<R>(data.size());
120  for (size_type i = 0; i < bin_edges.size() - 1; ++i)
121  {
122  prob[i] /= (static_cast<R>(bin_edges[i + 1] - bin_edges[i]) * n);
123  }
124  }
125 
126  return prob;
127  }
128 
129  } //detail

288  template <class E1, class E2, class E3>
289  inline auto histogram_bin_edges(E1&& data,
290  E2&& weights,
291  E3 left,
292  E3 right,
293  std::size_t bins = 10,
294  histogram_algorithm mode = histogram_algorithm::automatic)
295  {
296  // counter and return type
297  using size_type = common_size_type_t<std::decay_t<E1>, std::decay_t<E2>>;
298  using value_type = typename std::decay_t<E1>::value_type;
299  using weights_type = typename std::decay_t<E2>::value_type;
300 
301  // basic checks
302  // - rank
303  XTENSOR_ASSERT(data.dimension() == 1);
304  XTENSOR_ASSERT(weights.dimension() == 1);
305  // - size
306  XTENSOR_ASSERT(weights.size() == data.size());
307  // - bounds
308  XTENSOR_ASSERT(left <= right);
309  // - non-empty
310  XTENSOR_ASSERT(bins > std::size_t(0));
311 
312  // act on different modes
313  switch (mode)
314  {
315  // bins of equal width
316  case histogram_algorithm::automatic:
317  {
318  xt::xtensor<value_type, 1> bin_edges
319  = xt::linspace<value_type>(left, right, bins + 1);
320  return bin_edges;
321  }
322 
323  // bins of equal width
324  case histogram_algorithm::linspace:
325  {
326  xt::xtensor<value_type, 1> bin_edges
327  = xt::linspace<value_type>(left, right, bins + 1);
328  return bin_edges;
329  }
330 
331  // bins of logarithmically increasing size
332  case histogram_algorithm::logspace:
333  {
334  using rhs_value_type
335  = std::conditional_t<xtl::is_integral<value_type>::value, double, value_type>;
336 
337  xtensor<value_type, 1> bin_edges = xt::cast<value_type>(
338  xt::logspace<rhs_value_type>(std::log10(left), std::log10(right), bins + 1));
339 
340  // TODO: replace above with below after 'xsimd' fix
341  // xt::xtensor<value_type,1> bin_edges = xt::logspace<value_type>(
342  // std::log10(left), std::log10(right), bins+1);
343 
344  return bin_edges;
345  }
346 
347  // same amount of data in each bin
348  case histogram_algorithm::uniform:
349  {
350  // indices that sort "data"
351  auto sorter = xt::argsort(data);
352 
353  // histogram: all of equal 'height'
354  // - height
355  weights_type w
356  = xt::sum<weights_type>(weights)[0] / static_cast<weights_type>(bins);
357  // - apply to all bins
358  xt::xtensor<weights_type, 1> count = w * xt::ones<weights_type>({ bins });
359 
360  // take cumulative sum, to act as easy look-up
361  count = xt::cumsum(count);
362 
363  // edges
364  // - allocate
365  std::vector<size_t> shape = { bins + 1 };
366  xt::xtensor<value_type, 1> bin_edges = xtensor<value_type, 1>::from_shape(shape);
367  // - first/last edge
368  bin_edges[0] = left;
369  bin_edges[bins] = right;
370  // - cumulative weight
371  weights_type cum_weight = static_cast<weights_type>(0);
372  // - current bin
373  size_type ibin = 0;
374  // - loop to find interior bin-edges
375  for (size_type i = 0; i < weights.size(); ++i)
376  {
377  if (cum_weight >= count[ibin])
378  {
379  bin_edges[ibin + 1] = data[sorter[i]];
380  ++ibin;
381  }
382  cum_weight += weights[sorter[i]];
383  }
384  return bin_edges;
385  }
386 
387  // bins of equal width
388  default:
389  {
390  xt::xtensor<value_type, 1> bin_edges
391  = xt::linspace<value_type>(left, right, bins + 1);
392  return bin_edges;
393  }
394  }
395  }

496  template <class E1, class E2, XTL_REQUIRES(is_xexpression<std::decay_t<E2>>)>
497  inline auto bincount(E1&& data, E2&& weights, std::size_t minlength = 0)
498  {
499  using result_value_type = typename std::decay_t<E2>::value_type;
500  using input_value_type = typename std::decay_t<E1>::value_type;
501  using size_type = typename std::decay_t<E1>::size_type;
502 
503  static_assert(xtl::is_integral<typename std::decay_t<E1>::value_type>::value,
504  "Bincount data has to be integral type.");
505  XTENSOR_ASSERT(data.dimension() == 1);
506  XTENSOR_ASSERT(weights.dimension() == 1);
507 
508  std::array<input_value_type, 2> left_right;
509  left_right = xt::minmax(data)();
510 
511  if (left_right[0] < input_value_type(0))
512  {
513  XTENSOR_THROW(std::runtime_error,
514  "Data argument for bincount can only contain positive integers!");
515  }
516 
517  xt::xtensor<result_value_type, 1> res = xt::zeros<result_value_type>(
518  { (std::max)(minlength, std::size_t(left_right[1] + 1)) });
519 
520  for (size_type i = 0; i < data.size(); ++i)
521  {
522  res(data(i)) += weights(i);
523  }
524 
525  return res;
526  }

545  template <class E>
546  inline xt::xtensor<size_t, 1> bin_items(size_t N, E&& weights)
547  {
548  if (weights.size() <= std::size_t(1))
549  {
550  xt::xtensor<size_t, 1> n = N * xt::ones<size_t>({1});
551  return n;
552  }
553 
554  #ifdef XTENSOR_ENABLE_ASSERT
555  using value_type = typename std::decay_t<E>::value_type;
556 
557  XTENSOR_ASSERT(xt::all(weights >= static_cast<value_type>(0)));
558  XTENSOR_ASSERT(xt::sum(weights)() > static_cast<value_type>(0));
559  #endif
560 
561  xt::xtensor<double, 1> P = xt::cast<double>(weights) / static_cast<double>(xt::sum(weights)());
562  xt::xtensor<size_t, 1> n = xt::ceil(static_cast<double>(N) * P);
563 
564  if (xt::sum(n)() == N)
565  {
566  return n;
567  }
568 
569  xt::xtensor<size_t, 1> d = xt::zeros<size_t>(P.shape());
570  xt::xtensor<size_t, 1> sorter = xt::argsort(P);
571  sorter = xt::view(sorter, xt::range(P.size(), _, -1));
572  sorter = xt::view(sorter, xt::range(0, xt::sum(n)(0) - N));
573  xt::view(d, xt::keep(sorter)) = 1;
574  n -= d;
575 
576  return n;
577  }
