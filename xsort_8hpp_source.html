33  template <class E, class F>
34  inline void call_over_leading_axis(E& ev, F&& fct)
35  {
36  std::size_t n_iters = 1;
37  std::ptrdiff_t secondary_stride;
38 
39  if (ev.layout() == layout_type::row_major)
40  {
41  n_iters = std::accumulate(ev.shape().begin(), ev.shape().end() - 1,
42  std::size_t(1), std::multiplies<>());
43  secondary_stride = adjust_secondary_stride(ev.strides()[ev.dimension() - 2],
44  *(ev.shape().end() - 1));
45  }
46  else
47  {
48  n_iters = std::accumulate(ev.shape().begin() + 1, ev.shape().end(),
49  std::size_t(1), std::multiplies<>());
50  secondary_stride = adjust_secondary_stride(ev.strides()[1],
51  *(ev.shape().begin()));
52  }
53 
54  std::ptrdiff_t offset = 0;
55 
56  for (std::size_t i = 0; i < n_iters; ++i, offset += secondary_stride)
57  {
58  fct(ev.data() + offset, ev.data() + offset + secondary_stride);
59  }
60  }

62  template <class E>
63  inline std::size_t leading_axis(const E& e)
64  {
65  if (e.layout() == layout_type::row_major)
66  {
67  return e.dimension() - 1;
68  }
69  else if (e.layout() == layout_type::column_major)
70  {
71  return 0;
72  }
73  XTENSOR_THROW(std::runtime_error, "Layout not supported.");
74  }

76  // get permutations to transpose and reverse-transpose array
77  inline std::pair<dynamic_shape<std::size_t>, dynamic_shape<std::size_t>>
78  get_permutations(std::size_t dim, std::size_t ax, layout_type layout)
79  {
80  dynamic_shape<std::size_t> permutation(dim);
81  std::iota(permutation.begin(), permutation.end(), std::size_t(0));
82  permutation.erase(permutation.begin() + std::ptrdiff_t(ax));
83 
84  if (layout == layout_type::row_major)
85  {
86  permutation.push_back(ax);
87  }
88  else
89  {
90  permutation.insert(permutation.begin(), ax);
91  }
92 
93  // TODO find a more clever way to get reverse permutation?
94  dynamic_shape<std::size_t> reverse_permutation;
95  for (std::size_t i = 0; i < dim; ++i)
96  {
97  auto it = std::find(permutation.begin(), permutation.end(), i);
98  reverse_permutation.push_back(std::size_t(std::distance(permutation.begin(), it)));
99  }
100 
101  return std::make_pair(std::move(permutation), std::move(reverse_permutation));
102  }

104  template <class E, class R, class F>
105  inline auto run_lambda_over_axis(const E& e, R& res, std::size_t axis, F&& lambda)
106  {
107  if (axis != detail::leading_axis(e))
108  {
109  dynamic_shape<std::size_t> permutation, reverse_permutation;
110  std::tie(permutation, reverse_permutation) = get_permutations(e.dimension(), axis, e.layout());
111 
112  res = transpose(e, permutation);
113  detail::call_over_leading_axis(res, std::forward<F>(lambda));
114  res = transpose(res, reverse_permutation);
115  }
116  else
117  {
118  res = e;
119  detail::call_over_leading_axis(res, std::forward<F>(lambda));
120  }
121  }

272  template <class Ed, class Ei>
273  inline void argsort_over_leading_axis(const Ed& data, Ei& inds)
274  {
275  std::size_t n_iters = 1;
276  std::ptrdiff_t data_secondary_stride, inds_secondary_stride;
277 
278  if (data.layout() == layout_type::row_major)
279  {
280  n_iters = std::accumulate(data.shape().begin(), data.shape().end() - 1,
281  std::size_t(1), std::multiplies<>());
282  data_secondary_stride = data.strides()[data.dimension() - 2];
283  inds_secondary_stride = inds.strides()[inds.dimension() - 2];
284  }
285  else
286  {
287  n_iters = std::accumulate(data.shape().begin() + 1, data.shape().end(),
288  std::size_t(1), std::multiplies<>());
289  data_secondary_stride = data.strides()[1];
290  inds_secondary_stride = inds.strides()[1];
291  }
292 
293  auto ptr = data.data();
294  auto indices_ptr = inds.data();
295 
296  for (std::size_t i = 0; i < n_iters; ++i, ptr += data_secondary_stride, indices_ptr += inds_secondary_stride)
297  {
298  auto comp = [&ptr](std::size_t x, std::size_t y) {
299  return *(ptr + x) < *(ptr + y);
300  };
301  std::iota(indices_ptr, indices_ptr + inds_secondary_stride, 0);
302  std::sort(indices_ptr, indices_ptr + inds_secondary_stride, comp);
303  }
304  }

306  template <class E, class R = typename detail::linear_argsort_result_type<E>::type>
307  inline auto flatten_argsort_impl(const xexpression<E>& e)
308  {
309  const auto& de = e.derived_cast();
310 
311  auto cit = de.template begin<layout_type::row_major>();
312  using const_iterator = decltype(cit);
313  auto ad = xiterator_adaptor<const_iterator, const_iterator>(cit, cit, de.size());
314 
315  using result_type = R;
316  result_type result;
317  result.resize({de.size()});
318  auto comp = [&ad](std::size_t x, std::size_t y) {
319  return ad[x] < ad[y];
320  };
321  std::iota(result.begin(), result.end(), 0);
322  std::sort(result.begin(), result.end(), comp);
323 
324  return result;
325  }
326  }

345  template <class E>
346  inline auto argsort(const xexpression<E>& e, std::ptrdiff_t axis = -1)
347  {
348  using eval_type = typename detail::sort_eval_type<E>::type;
349  using result_type = typename detail::argsort_result_type<eval_type>::type;
350 
351  const auto& de = e.derived_cast();
352 
353  std::size_t ax = normalize_axis(de.dimension(), axis);
354 
355  if (de.dimension() == 1)
356  {
357  return detail::flatten_argsort_impl<E, result_type>(e);
358  }
359 
360  if (ax != detail::leading_axis(de))
361  {
362  dynamic_shape<std::size_t> permutation, reverse_permutation;
363  std::tie(permutation, reverse_permutation) = detail::get_permutations(de.dimension(), ax, de.layout());
364 
365  eval_type ev = transpose(de, permutation);
366  result_type res = result_type::from_shape(ev.shape());
367  detail::argsort_over_leading_axis(ev, res);
368  res = transpose(res, reverse_permutation);
369  return res;
370  }
371  else
372  {
373  result_type res = result_type::from_shape(de.shape());
374  detail::argsort_over_leading_axis(de, res);
375  return res;

408  template <class E, class C, class R = detail::flatten_sort_result_type_t<E>,
409  class = std::enable_if_t<!xtl::is_integral<C>::value, int>>
410  inline R partition(const xexpression<E>& e, const C& kth_container, placeholders::xtuph /*ax*/)
411  {
412  const auto& de = e.derived_cast();
413 
414  R ev = R::from_shape({ de.size() });
415  C kth_copy = kth_container;
416  if (kth_copy.size() > 1)
417  {
418  std::sort(kth_copy.begin(), kth_copy.end());
419  }
420 
421  std::copy(de.storage_cbegin(), de.storage_cend(), ev.storage_begin()); // flatten
422  std::size_t k_last = kth_copy.back();
423  std::nth_element(ev.storage_begin(), ev.storage_begin() + k_last, ev.storage_end());
424 
425  for (auto it = (kth_copy.rbegin() + 1); it != kth_copy.rend(); ++it)
426  {
427  std::nth_element(ev.storage_begin(), ev.storage_begin() + *it, ev.storage_begin() + k_last);
428  k_last = *it;
429  }
430 
431  return ev;
432  }

446  template <class E, class C, class = std::enable_if_t<!xtl::is_integral<C>::value, int>>
447  inline auto partition(const xexpression<E>& e, const C& kth_container, std::ptrdiff_t axis = -1)
448  {
449  using eval_type = typename detail::sort_eval_type<E>::type;
450 
451  const auto& de = e.derived_cast();
452 
453  if (de.dimension() == 1)
454  {
455  return partition<E, C, eval_type>(de, kth_container, xnone());
456  }
457 
458  C kth_copy = kth_container;
459  if (kth_copy.size() > 1)
460  {
461  std::sort(kth_copy.begin(), kth_copy.end());
462  }
463 
464  std::size_t ax = normalize_axis(de.dimension(), axis);
465 
466  eval_type res;
467 
468  std::size_t kth = kth_copy.back();
469 
470  dynamic_shape<std::size_t> permutation, reverse_permutation;
471  bool is_leading_axis = (ax == detail::leading_axis(de));
472 
473  if (!is_leading_axis)
474  {
475  std::tie(permutation, reverse_permutation) = detail::get_permutations(de.dimension(), ax, de.layout());
476  res = transpose(de, permutation);
477  }
478  else
479  {
480  res = de;
481  }
482 
483  auto lambda = [&kth](auto begin, auto end) {
484  std::nth_element(begin, begin + kth, end);
485  };
486  detail::call_over_leading_axis(res, lambda);
487 
488  for (auto it = kth_copy.rbegin() + 1; it != kth_copy.rend(); ++it)
489  {
490  kth = *it;
491  detail::call_over_leading_axis(res, lambda);
492  }
493 
494  if (!is_leading_axis)
495  {
496  res = transpose(res, reverse_permutation);
497  }
498 
499  return res;
500  }

600  }
601  else
602  {
603  n_iters = std::accumulate(data.shape().begin() + 1, data.shape().end(),
604  std::size_t(1), std::multiplies<>());
605  data_secondary_stride = data.strides()[1];
606  inds_secondary_stride = inds.strides()[1];

700  template <class E>
701  inline typename std::decay_t<E>::value_type median(E&& e)
702  {
703  using value_type = typename std::decay_t<E>::value_type;
704  auto sz = e.size();
705  if (sz % 2 == 0)

801  using value_type = typename E::value_type;

803  using result_shape_type = typename result_type::shape_type;
804 
805  if (e.dimension() == 1)
806  {
807  return arg_func_impl<L>(e, std::forward<F>(cmp));
808  }

810  result_shape_type alt_shape;
811  xt::resize_container(alt_shape, e.dimension() - 1);
812 
813  // Excluding copy, copy all of shape except for axis
814  std::copy(e.shape().cbegin(), e.shape().cbegin() + std::ptrdiff_t(axis), alt_shape.begin());

816 
817  result_type result = result_type::from_shape(std::move(alt_shape));
818  auto result_iter = result.template begin<L>();

821  std::size_t idx = 0;
822  value_type val = *begin;
823  ++begin;
824  for (std::size_t i = 1; begin != end; ++begin, ++i)
825  {
826  if (cmp(*begin, val))
827  {
828  val = *begin;
829  idx = i;
830  }
831  }
832  *result_iter = idx;
833  ++result_iter;
834  };

845  }
846  else

849  detail::call_over_leading_axis(input, arg_func_lambda);
