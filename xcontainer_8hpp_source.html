1030  inline void xstrided_container<D>::reshape_impl(S&& _shape, std::true_type /* is signed */, layout_type layout)
1031  {

1033  auto new_size = compute_size(_shape);
1034  if (this->size() % new_size)
1035  {
1036  XTENSOR_THROW(std::runtime_error, "Negative axis size cannot be inferred. Shape mismatch.");
1037  }
1038  std::decay_t<S> shape = _shape;
1039  tmp_value_type accumulator = 1;
1040  std::size_t neg_idx = 0;
1041  std::size_t i = 0;
1042  for(auto it = shape.begin(); it != shape.end(); ++it, i++)
1043  {
1044  auto&& dim = *it;
1045  if(dim < 0)
1046  {
1047  XTENSOR_ASSERT(dim == -1 && !neg_idx);
1048  neg_idx = i;
1049  }
1050  accumulator *= dim;
1051  }
1052  if(accumulator < 0)
1053  {
1054  shape[neg_idx] = static_cast<tmp_value_type>(this->size()) / std::abs(accumulator);
1055  }
1056  else if(this->size() != new_size)
1057  {

1059  }

1062  resize_container(m_strides, m_shape.size());
