485  template<class S, class R>
486  inline void xrepeat_stepper<S, R>::step(size_type dim, size_type steps_to_go)
487  {
488  if (m_positions[dim] + steps_to_go >= m_shape[dim])
489  {
490  const auto next_positions = get_next_positions(dim, steps_to_go);
491  if (next_positions[dim] > m_positions[dim])
492  {
493  make_step(dim, next_positions[dim] - m_positions[dim]);
494  }
495  else
496  {
497  make_step_back(dim, m_positions[dim] - next_positions[dim]);
498  }
499  for (size_type d = 0; d < dim; ++d)
500  {
501  make_step(d, next_positions[d] - m_positions[d]);
502  }
503  }
504  else
505  {
506  make_step(dim, steps_to_go);
507  }
508  }

510  template<class S, class R>
511  inline void xrepeat_stepper<S, R>::step_back(size_type dim, size_type steps_to_go)
512  {
513  if (m_positions[dim] < steps_to_go)
514  {
515  const auto next_positions = get_next_positions_back(dim, steps_to_go);
516  if (next_positions[dim] < m_positions[dim])
517  {
518  make_step_back(dim, m_positions[dim] - next_positions[dim]);
519  }
520  else
521  {
522  make_step(dim, next_positions[dim] - m_positions[dim]);
523  }
524  for (size_type d = 0; d < dim; ++d)
525  {
526  make_step_back(d, m_positions[d] - next_positions[d]);
527  }
528  }
529  else
530  {
531  make_step_back(dim, steps_to_go);
532  }
533  }

593  template<class S, class R>
594  inline void xrepeat_stepper<S, R>::make_step(size_type dim, size_type steps_to_go)
595  {
596  if (steps_to_go > 0)
597  {
598  if (dim == m_repeating_axis)
599  {
600  size_type subposition = m_subposition;
601  m_repeating_steps += static_cast<std::ptrdiff_t>(steps_to_go);
602  while (m_repeating_steps >= static_cast<ptrdiff_t>(m_repeats[subposition]))
603  {
604  m_repeating_steps -= static_cast<ptrdiff_t>(m_repeats[subposition]);
605  ++subposition;
606  }
607  m_substepper.step(dim, subposition - m_subposition);
608  m_subposition = subposition;
609  }
610  else
611  {
612  m_substepper.step(dim, steps_to_go);
613  }
614  m_positions[dim] += steps_to_go;
615  }
616  }

618  template<class S, class R>
619  inline void xrepeat_stepper<S, R>::make_step_back(size_type dim, size_type steps_to_go)
620  {
621  if (steps_to_go > 0)
622  {
623  if (dim == m_repeating_axis)
624  {
625  size_type subposition = m_subposition;
626  m_repeating_steps -= static_cast<std::ptrdiff_t>(steps_to_go);
627  while (m_repeating_steps < 0)
628  {
629  --subposition;
630  m_repeating_steps += static_cast<ptrdiff_t>(m_repeats[subposition]);
631  }
632  m_substepper.step_back(dim, m_subposition - subposition);
633  m_subposition = subposition;
634  }
635  else
636  {
637  m_substepper.step_back(dim, steps_to_go);
638  }
639  m_positions[dim] -= steps_to_go;
640  }

643  template<class S, class R>
644  inline auto xrepeat_stepper<S, R>::get_next_positions(size_type dim, size_type steps_to_go) const -> std::vector<size_type>
645  {
646  size_type next_position_for_dim = m_positions[dim] + steps_to_go;
647  if (dim > 0)
648  {
649  size_type steps_in_previous_dim = 0;
650  while (next_position_for_dim >= m_shape[dim])
651  {
652  next_position_for_dim -= m_shape[dim];
653  ++steps_in_previous_dim;
654  }
655  if (steps_in_previous_dim > 0)
656  {
657  auto next_positions = get_next_positions(dim - 1, steps_in_previous_dim);
658  next_positions[dim] = next_position_for_dim;
659  return next_positions;
660  }
661  }
662  std::vector<size_type> next_positions = m_positions;
663  next_positions[dim] = next_position_for_dim;
664  return next_positions;
665  }
