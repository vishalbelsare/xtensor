594  template <class CT, class S, layout_type L, class FST>
595  template <class T, class... Args>
596  inline auto xdynamic_view<CT, S, L, FST>::adjust_offset_impl(offset_type offset, size_type idx_offset, T idx, Args... args) const noexcept
597  -> offset_type
598  {
599  offset_type sl_offset = xtl::visit([idx](const auto& sl) {
600  using type = typename std::decay_t<decltype(sl)>::size_type;

618  auto dst = std::distance(first, last);
619  offset_type dim = static_cast<offset_type>(dimension());
620  offset_type loop_offset = dst < dim ? dim - dst : offset_type(0);
621  offset_type idx_offset = dim < dst ? dst - dim : offset_type(0);
622  offset_type res = offset;

627  res += sl_offset * m_adj_strides[static_cast<std::size_t>(i)];
628  }
