106  inline layout_type transpose_layout_noexcept(layout_type l) noexcept
107  {
108  layout_type result = l;
109  if (l == layout_type::row_major)
110  {
111  result = layout_type::column_major;
112  }
113  else if (l == layout_type::column_major)
114  {
115  result = layout_type::row_major;
116  }
117  return result;
118  }
119 
120  inline layout_type transpose_layout(layout_type l)
121  {
122  if (l != layout_type::row_major && l != layout_type::column_major)
123  {
124  XTENSOR_THROW(transpose_error, "cannot compute transposed layout of dynamic layout");
125  }
126  return transpose_layout_noexcept(l);
127  }

129  template <class E, class S>
130  inline auto transpose_impl(E&& e, S&& permutation, check_policy::none)
131  {
132  if (sequence_size(permutation) != e.dimension())
133  {
134  XTENSOR_THROW(transpose_error, "Permutation does not have the same size as shape");
135  }
136 
137  // permute stride and shape
138  using shape_type = xindex_type_t<typename std::decay_t<E>::shape_type>;
139  shape_type temp_shape;
140  resize_container(temp_shape, e.shape().size());
141 
142  using strides_type = get_strides_t<shape_type>;
143  strides_type temp_strides;
144  resize_container(temp_strides, e.strides().size());
145 
146  using size_type = typename std::decay_t<E>::size_type;
147  for (std::size_t i = 0; i < e.shape().size(); ++i)
148  {
149  if (std::size_t(permutation[i]) >= e.dimension())
150  {
151  XTENSOR_THROW(transpose_error, "Permutation contains wrong axis");
152  }
153  size_type perm = static_cast<size_type>(permutation[i]);
154  temp_shape[i] = e.shape()[perm];
155  temp_strides[i] = e.strides()[perm];
156  }
157 
158  layout_type new_layout = layout_type::dynamic;
159  if (std::is_sorted(std::begin(permutation), std::end(permutation)))
160  {
161  // keep old layout
162  new_layout = e.layout();
163  }
164  else if (std::is_sorted(std::begin(permutation), std::end(permutation), std::greater<>()))
165  {
166  new_layout = transpose_layout_noexcept(e.layout());
167  }
168 
169  return strided_view(std::forward<E>(e), std::move(temp_shape), std::move(temp_strides), get_offset<XTENSOR_DEFAULT_LAYOUT>(e), new_layout);
170  }

172  template <class E, class S>
173  inline auto transpose_impl(E&& e, S&& permutation, check_policy::full)
174  {
175  // check if axis appears twice in permutation
176  for (std::size_t i = 0; i < sequence_size(permutation); ++i)
177  {
178  for (std::size_t j = i + 1; j < sequence_size(permutation); ++j)
179  {
180  if (permutation[i] == permutation[j])
181  {
182  XTENSOR_THROW(transpose_error, "Permutation contains axis more than once");
183  }
184  }
185  }
186  return transpose_impl(std::forward<E>(e), std::forward<S>(permutation), check_policy::none());
187  }

195  template <class E, class S, class X, std::enable_if_t<!has_data_interface<std::decay_t<E>>::value>* = nullptr>
196  inline void compute_transposed_strides(E&&, const S& shape, X& strides)
197  {
198  // In the case where E does not have a data interface, the transposition
199  // makes use of a flat storage adaptor that has layout XTENSOR_DEFAULT_TRAVERSAL
200  // which should be the one inverted.
201  layout_type l = transpose_layout(XTENSOR_DEFAULT_TRAVERSAL);
202  compute_strides(shape, l, strides);
203  }

210  template <class E>
211  inline auto transpose(E&& e) noexcept
212  {
213  using shape_type = xindex_type_t<typename std::decay_t<E>::shape_type>;
214  shape_type shape;
215  resize_container(shape, e.shape().size());
216  std::copy(e.shape().crbegin(), e.shape().crend(), shape.begin());
217 
218  get_strides_t<shape_type> strides;
219  resize_container(strides, e.shape().size());
220  detail::compute_transposed_strides(e, shape, strides);
221 
222  layout_type new_layout = detail::transpose_layout_noexcept(e.layout());
223 
224  return strided_view(std::forward<E>(e), std::move(shape), std::move(strides), detail::get_offset<XTENSOR_DEFAULT_TRAVERSAL>(e), new_layout);
225  }

337  template <class E>
338  inline auto trim_zeros(E&& e, const std::string& direction)
339  {
340  XTENSOR_ASSERT_MSG(e.dimension() == 1, "Dimension for trim_zeros has to be 1.");
341 
342  std::ptrdiff_t begin = 0, end = static_cast<std::ptrdiff_t>(e.size());
343 
344  auto find_fun = [](const auto& i) {
345  return i != 0;
346  };
347 
348  if (direction.find("f") != std::string::npos)
349  {
350  begin = std::find_if(e.cbegin(), e.cend(), find_fun) - e.cbegin();
351  }
352 
353  if (direction.find("b") != std::string::npos && begin != end)
354  {
355  end -= std::find_if(e.crbegin(), e.crend(), find_fun) - e.crbegin();
356  }
357 
358  return strided_view(std::forward<E>(e), { range(begin, end) });
359  }

372  template <class E>
373  inline auto squeeze(E&& e)
374  {
375  dynamic_shape<std::size_t> new_shape;
376  dynamic_shape<std::ptrdiff_t> new_strides;
377  std::copy_if(e.shape().cbegin(), e.shape().cend(), std::back_inserter(new_shape),
378  [](std::size_t i) { return i != 1; });
379  decltype(auto) old_strides = detail::get_strides<XTENSOR_DEFAULT_LAYOUT>(e);
380  std::copy_if(old_strides.cbegin(), old_strides.cend(), std::back_inserter(new_strides),
381  [](std::ptrdiff_t i) { return i != 0; });
382 
383  return strided_view(std::forward<E>(e), std::move(new_shape), std::move(new_strides), 0, e.layout());
384  }

388  template <class E, class S>
389  inline auto squeeze_impl(E&& e, S&& axis, check_policy::none)
390  {
391  std::size_t new_dim = e.dimension() - axis.size();
392  dynamic_shape<std::size_t> new_shape(new_dim);
393  dynamic_shape<std::ptrdiff_t> new_strides(new_dim);
394 
395  decltype(auto) old_strides = detail::get_strides<XTENSOR_DEFAULT_LAYOUT>(e);
396 
397  for (std::size_t i = 0, ix = 0; i < e.dimension(); ++i)
398  {
399  if (axis.cend() == std::find(axis.cbegin(), axis.cend(), i))
400  {
401  new_shape[ix] = e.shape()[i];
402  new_strides[ix++] = old_strides[i];
403  }
404  }
405 
406  return strided_view(std::forward<E>(e), std::move(new_shape), std::move(new_strides), 0, e.layout());
407  }

562  template <class E>
563  inline auto split(E& e, std::size_t n, std::size_t axis)
564  {
565  if (axis >= e.dimension())
566  {
567  XTENSOR_THROW(std::runtime_error, "Split along axis > dimension.");
568  }
569 
570  std::size_t ax_sz = e.shape()[axis];
571  xstrided_slice_vector sv(e.dimension(), all());
572  std::size_t step = ax_sz / n;
573  std::size_t rest = ax_sz % n;
574 
575  if (rest)
576  {
577  XTENSOR_THROW(std::runtime_error, "Split does not result in equal division.");
578  }
579 
580  std::vector<decltype(strided_view(e, sv))> result;
581  for (std::size_t i = 0; i < n; ++i)
582  {
583  sv[axis] = range(i * step, (i + 1) * step);
584  result.emplace_back(strided_view(e, sv));
585  }
586  return result;
587  }

632  inline auto flip(E&& e, std::size_t axis)
633  {

635 
636  shape_type shape;
637  resize_container(shape, e.shape().size());

642  resize_container(strides, old_strides.size());

752  template<class E>
753  inline auto roll(E&& e, std::ptrdiff_t shift)
754  {
755  auto cpy = empty_like(e);
756  auto flat_size = std::accumulate(cpy.shape().begin(), cpy.shape().end(), 1L, std::multiplies<std::size_t>());
757  while(shift < 0)
758  {
759  shift += flat_size;
760  }
761 
762  shift %= flat_size;
763  std::copy(e.begin(), e.end() - shift,
764  std::copy(e.end() - shift, e.end(), cpy.begin()));
765 
766  return cpy;
767  }

769  namespace detail
770  {
775  template < class To, class From, class S>
776  To roll(To to, From from, std::ptrdiff_t shift, std::size_t axis, S const& shape, std::size_t M)
777  {
778  std::ptrdiff_t dim = std::ptrdiff_t(shape[M]);
779  std::ptrdiff_t offset = std::accumulate(shape.begin() + M + 1, shape.end(), std::ptrdiff_t(1), std::multiplies<std::ptrdiff_t>());
780  if(shape.size() == M + 1)
781  {
782  if (axis == M)
783  {
784  const auto split = from + (dim - shift) * offset;
785  for(auto iter = split, end = from + dim * offset; iter != end; iter += offset, ++to)
786  {
787  *to = *iter;
788  }
789  for(auto iter = from, end = split; iter != end; iter += offset, ++to)
790  {
791  *to = *iter;
792  }
793  }
794  else
795  {
796  for(auto iter = from, end = from + dim * offset; iter != end; iter += offset, ++to)
797  {
798  *to = *iter;
799  }
800  }
801  }
802  else
803  {
804  if (axis == M)
805  {
806  const auto split = from + (dim - shift) * offset;
807  for(auto iter = split, end = from + dim * offset; iter != end; iter += offset)
808  {
809  to = roll(to, iter, shift, axis, shape, M + 1);
810  }
811  for(auto iter = from, end = split; iter != end; iter += offset)
812  {
813  to = roll(to, iter, shift, axis, shape, M + 1);
814  }
815  }
