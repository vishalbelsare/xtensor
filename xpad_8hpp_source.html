72  template <class E,
73  class S = typename std::decay_t<E>::size_type,
74  class V = typename std::decay_t<E>::value_type>
75  inline auto pad(E&& e,
76  const std::vector<std::vector<S>>& pad_width,
77  pad_mode mode = pad_mode::constant,
78  V constant_value = 0)
79  {
80  XTENSOR_ASSERT(detail::check_pad_width(pad_width, e.shape()));
81 
82  using size_type = typename std::decay_t<E>::size_type;
83  using return_type = temporary_type_t<E>;
84 
85  // place the original array in the center
86 
87  auto new_shape = e.shape();
88  xt::xstrided_slice_vector sv;
89  sv.reserve(e.shape().size());
90  for (size_type axis = 0; axis < e.shape().size(); ++axis)
91  {
92  size_type nb = static_cast<size_type>(pad_width[axis][0]);
93  size_type ne = static_cast<size_type>(pad_width[axis][1]);
94  size_type ns = nb + e.shape(axis) + ne;
95  new_shape[axis] = ns;
96  sv.push_back(xt::range(nb, nb + e.shape(axis)));
97  }
98 
99  if (mode == pad_mode::constant)
100  {
101  return_type out(new_shape, constant_value);
102  xt::strided_view(out, sv) = e;
103  return out;
104  }
105 
106  return_type out(new_shape);
107  xt::strided_view(out, sv) = e;
108 
109  // construct padded regions based on original image
110 
111  xt::xstrided_slice_vector svs(e.shape().size(), xt::all());
112  xt::xstrided_slice_vector svt(e.shape().size(), xt::all());
113 
114  for (size_type axis = 0; axis < e.shape().size(); ++axis)
115  {
116  size_type nb = static_cast<size_type>(pad_width[axis][0]);
117  size_type ne = static_cast<size_type>(pad_width[axis][1]);
118 
119  if (nb > static_cast<size_type>(0))
120  {
121  svt[axis] = xt::range(0, nb);
122 
123  if (mode == pad_mode::wrap || mode == pad_mode::periodic)
124  {
125  XTENSOR_ASSERT(nb <= e.shape(axis));
126  svs[axis] = xt::range(e.shape(axis), nb+e.shape(axis));
127  }
128  else if (mode == pad_mode::symmetric)
129  {
130  XTENSOR_ASSERT(nb <= e.shape(axis));
131  svs[axis] = xt::range(2*nb-1, nb-1, -1);
132  }
133  else if (mode == pad_mode::reflect)
134  {
135  XTENSOR_ASSERT(nb <= e.shape(axis) - 1);
136  svs[axis] = xt::range(2*nb, nb, -1);
137  }
138 
139  xt::strided_view(out, svt) = xt::strided_view(out, svs);
140  }
141 
142  if (ne > static_cast<size_type>(0))
143  {
144  svt[axis] = xt::range(out.shape(axis)-ne, out.shape(axis));
145 
146  if (mode == pad_mode::wrap || mode == pad_mode::periodic)
147  {
148  XTENSOR_ASSERT(ne <= e.shape(axis));
149  svs[axis] = xt::range(nb, nb+ne);
150  }
151  else if (mode == pad_mode::symmetric)
152  {
153  XTENSOR_ASSERT(ne <= e.shape(axis));
154  if (ne == nb + e.shape(axis))
155  {
156  svs[axis] = xt::range(nb+e.shape(axis)-1, _, -1);
157  }
158  else
159  {
160  svs[axis] = xt::range(nb+e.shape(axis)-1, nb+e.shape(axis)-ne-1, -1);
161  }
162  }
163  else if (mode == pad_mode::reflect)
164  {
165  XTENSOR_ASSERT(ne <= e.shape(axis) - 1);
166  if (ne == nb + e.shape(axis) - 1)
167  {
168  svs[axis] = xt::range(nb+e.shape(axis)-2, _, -1);
169  }
170  else
171  {
172  svs[axis] = xt::range(nb+e.shape(axis)-2, nb+e.shape(axis)-ne-2, -1);
173  }
174  }
175 
176  xt::strided_view(out, svt) = xt::strided_view(out, svs);
177  }
178 
179  svs[axis] = xt::all();
180  svt[axis] = xt::all();
181  }
182 
183  return out;
184  }

235  template <class E, class S>
236  inline auto tile(E&& e, const S& reps)
237  {
238  using size_type = typename std::decay_t<E>::size_type;
239 
240  using return_type = temporary_type_t<E>;
241 
242  XTENSOR_ASSERT(e.shape().size() == reps.size());
243 
244  using new_shape_type = typename return_type::shape_type;
245  auto new_shape = xtl::make_sequence<new_shape_type>(e.shape().size());
246 
247  xt::xstrided_slice_vector sv(reps.size());
248 
249  for (size_type axis = 0; axis < reps.size(); ++axis)
250  {
251  new_shape[axis] = e.shape(axis) * reps[axis];
252  sv[axis] = xt::range(0, e.shape(axis));
253  }
254  return_type out(new_shape);
255 
256  xt::strided_view(out, sv) = e;
257 
258  xt::xstrided_slice_vector svs(e.shape().size(), xt::all());
259  xt::xstrided_slice_vector svt(e.shape().size(), xt::all());
260 
261  for (size_type axis = 0; axis < e.shape().size(); ++axis)
262  {
263  for (size_type i = 1; i < static_cast<size_type>(reps[axis]); ++i)
264  {
265  svs[axis] = xt::range(0, e.shape(axis));
266  svt[axis] = xt::range(i * e.shape(axis), (i + 1) * e.shape(axis));
267  xt::strided_view(out, svt) = xt::strided_view(out, svs);
268  svs[axis] = xt::all();
269  svt[axis] = xt::all();
270  }
271  }
272 
273  return out;
274  }
