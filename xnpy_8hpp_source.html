61  inline void read_magic(std::istream& istream,
62  unsigned char* v_major,
63  unsigned char* v_minor)
64  {
65  std::unique_ptr<char[]> buf(new char[magic_string_length + 2]);
66  istream.read(buf.get(), magic_string_length + 2);
67 
68  if (!istream)
69  {
70  XTENSOR_THROW(std::runtime_error, "io error: failed reading file");
71  }
72 
73  for (std::size_t i = 0; i < magic_string_length; i++)
74  {
75  if (buf[i] != magic_string[i])
76  {
77  XTENSOR_THROW(std::runtime_error, "this file do not have a valid npy format.");
78  }
79  }
80 
81  *v_major = static_cast<unsigned char>(buf[magic_string_length]);
82  *v_minor = static_cast<unsigned char>(buf[magic_string_length + 1]);
83  }

85  template <class T>
86  inline char map_type()
87  {
88  if (std::is_same<T, float>::value) return 'f';
89  if (std::is_same<T, double>::value) return 'f';
90  if (std::is_same<T, long double>::value) return 'f';
91 
92  if (std::is_same<T, char>::value) return 'i';
93  if (std::is_same<T, signed char>::value) return 'i';
94  if (std::is_same<T, short>::value) return 'i';
95  if (std::is_same<T, int>::value) return 'i';
96  if (std::is_same<T, long>::value) return 'i';
97  if (std::is_same<T, long long>::value) return 'i';
98 
99  if (std::is_same<T, unsigned char>::value) return 'u';
100  if (std::is_same<T, unsigned short>::value) return 'u';
101  if (std::is_same<T, unsigned int>::value) return 'u';
102  if (std::is_same<T, unsigned long>::value) return 'u';
103  if (std::is_same<T, unsigned long long>::value) return 'u';
104 
105  if (std::is_same<T, bool>::value) return 'b';
106 
107  if (std::is_same<T, std::complex<float>>::value) return 'c';
108  if (std::is_same<T, std::complex<double>>::value) return 'c';
109  if (std::is_same<T, std::complex<long double>>::value) return 'c';
110 
111  XTENSOR_THROW(std::runtime_error, "Type not known.");
112  }

114  template <class T>
115  inline char get_endianess()
116  {
117  constexpr char little_endian_char = '<';
118  constexpr char big_endian_char = '>';
119  constexpr char no_endian_char = '|';
120 
121  if(sizeof(T) <= sizeof(char))
122  {
123  return no_endian_char;
124  }
125 
126  switch(xtl::endianness())
127  {
128  case xtl::endian::little_endian:
129  return little_endian_char;
130  case xtl::endian::big_endian:
131  return big_endian_char;
132  default:
133  return no_endian_char;
134  }
135  }

190  inline void parse_header(std::string header, std::string& descr,
191  bool* fortran_order,
192  std::vector<std::size_t>& shape)
193  {
194  // The first 6 bytes are a magic string: exactly "x93NUMPY".
195  //
196  // The next 1 byte is an unsigned byte: the major version number of the file
197  // format, e.g. x01.
198  //
199  // The next 1 byte is an unsigned byte: the minor version number of the file
200  // format, e.g. x00. Note: the version of the file format is not tied to the
201  // version of the numpy package.
202  //
203  // The next 2 bytes form a little-endian unsigned short int: the length of the
204  // header data HEADER_LEN.
205  //
206  // The next HEADER_LEN bytes form the header data describing the array's
207  // format. It is an ASCII string which contains a Python literal expression of
208  // a dictionary. It is terminated by a newline ('n') and padded with spaces
209  // ('x20') to make the total length of the magic string + 4 + HEADER_LEN be
210  // evenly divisible by 16 for alignment purposes.
211  //
212  // The dictionary contains three keys:
213  //
214  // "descr" : dtype.descr
215  // An object that can be passed as an argument to the numpy.dtype()
216  // constructor to create the array's dtype.
217  // "fortran_order" : bool
218  // Whether the array data is Fortran-contiguous or not. Since
219  // Fortran-contiguous arrays are a common form of non-C-contiguity, we allow
220  // them to be written directly to disk for efficiency.
221  // "shape" : tuple of int
222  // The shape of the array.
223  // For repeatability and readability, this dictionary is formatted using
224  // pprint.pformat() so the keys are in alphabetic order.
225 
226  // remove trailing newline
227  if (header.back() != '\n')
228  {
229  XTENSOR_THROW(std::runtime_error, "invalid header");
230  }
231  header.pop_back();
232 
233  // remove all whitespaces
234  header.erase(std::remove(header.begin(), header.end(), ' '), header.end());
235 
236  // unwrap dictionary
237  header = unwrap_s(header, '{', '}');
238 
239  // find the positions of the 3 dictionary keys
240  std::size_t keypos_descr = header.find("'descr'");
241  std::size_t keypos_fortran = header.find("'fortran_order'");
242  std::size_t keypos_shape = header.find("'shape'");
243 
244  // make sure all the keys are present
245  if (keypos_descr == std::string::npos)
246  {
247  XTENSOR_THROW(std::runtime_error, "missing 'descr' key");
248  }
249  if (keypos_fortran == std::string::npos)
250  {
251  XTENSOR_THROW(std::runtime_error, "missing 'fortran_order' key");
252  }
253  if (keypos_shape == std::string::npos)
254  {
255  XTENSOR_THROW(std::runtime_error, "missing 'shape' key");
256  }
257 
258  // Make sure the keys are in order.
259  // Note that this violates the standard, which states that readers *must* not
260  // depend on the correct order here.
261  // TODO: fix
262  if (keypos_descr >= keypos_fortran || keypos_fortran >= keypos_shape)
263  {
264  XTENSOR_THROW(std::runtime_error, "header keys in wrong order");
265  }
266 
267  // get the 3 key-value pairs
268  std::string keyvalue_descr;
269  keyvalue_descr = header.substr(keypos_descr, keypos_fortran - keypos_descr);
270  pop_char(keyvalue_descr, ',');
271 
272  std::string keyvalue_fortran;
273  keyvalue_fortran = header.substr(keypos_fortran, keypos_shape - keypos_fortran);
274  pop_char(keyvalue_fortran, ',');
275 
276  std::string keyvalue_shape;
277  keyvalue_shape = header.substr(keypos_shape, std::string::npos);
278  pop_char(keyvalue_shape, ',');
279 
280  // get the values (right side of `:')
281  std::string descr_s = get_value_from_map(keyvalue_descr);
282  std::string fortran_s = get_value_from_map(keyvalue_fortran);
283  std::string shape_s = get_value_from_map(keyvalue_shape);
284 
285  parse_typestring(descr_s);
286  descr = unwrap_s(descr_s, '\'', '\'');
287 
288  // convert literal Python bool to C++ bool
289  if (fortran_s == "True")
290  {
291  *fortran_order = true;
292  }
293  else if (fortran_s == "False")
294  {
295  *fortran_order = false;
296  }
297  else
298  {
299  XTENSOR_THROW(std::runtime_error, "invalid fortran_order value");
300  }
301 
302  // parse the shape Python tuple ( x, y, z,)
303 
304  // first clear the vector
305  shape.clear();
306  shape_s = unwrap_s(shape_s, '(', ')');
307 
308  // a tokenizer would be nice...
309  std::size_t pos = 0;
310  for (;;)
311  {
312  std::size_t pos_next = shape_s.find_first_of(',', pos);
313  std::string dim_s;
314 
315  if (pos_next != std::string::npos)
316  {
317  dim_s = shape_s.substr(pos, pos_next - pos);
318  }
319  else
320  {
321  dim_s = shape_s.substr(pos);
322  }
323 
324  if (dim_s.length() == 0)
325  {
326  if (pos_next != std::string::npos)
327  {
328  XTENSOR_THROW(std::runtime_error, "invalid shape");
329  }
330  }
331  else
332  {
333  std::stringstream ss;
334  ss << dim_s;
335  std::size_t tmp;
336  ss >> tmp;
337  shape.push_back(tmp);
338  }
339 
340  if (pos_next != std::string::npos)
341  {
342  pos = ++pos_next;
343  }
344  else
345  {
346  break;
347  }
348  }
349  }

351  template <class O, class S>
352  inline void write_header(O& out, const std::string& descr,
353  bool fortran_order, const S& shape)
354  {
355  std::ostringstream ss_header;
356  std::string s_fortran_order;
357  if (fortran_order)
358  {
359  s_fortran_order = "True";
360  }
361  else
362  {
363  s_fortran_order = "False";
364  }
365 
366  std::string s_shape;
367  std::ostringstream ss_shape;
368  ss_shape << "(";
369  for (auto shape_it = std::begin(shape); shape_it != std::end(shape); ++shape_it)
370  {
371  ss_shape << *shape_it << ", ";
372  }
373  s_shape = ss_shape.str();
374  if (xtl::sequence_size(shape) > 1)
375  {
376  s_shape = s_shape.erase(s_shape.size() - 2);
377  }
378  else if (xtl::sequence_size(shape) == 1)
379  {
380  s_shape = s_shape.erase(s_shape.size() - 1);
381  }
382  s_shape += ")";
383 
384  ss_header << "{'descr': '" << descr
385  << "', 'fortran_order': " << s_fortran_order
386  << ", 'shape': " << s_shape << ", }";
387 
388  std::size_t header_len_pre = ss_header.str().length() + 1;
389  std::size_t metadata_len = magic_string_length + 2 + 2 + header_len_pre;
390 
391  unsigned char version[2] = {1, 0};
392  if (metadata_len >= 255 * 255)
393  {
394  metadata_len = magic_string_length + 2 + 4 + header_len_pre;
395  version[0] = 2;
396  version[1] = 0;
397  }
398  std::size_t padding_len = 64 - (metadata_len % 64);
399  std::string padding(padding_len, ' ');
400  ss_header << padding;
401  ss_header << std::endl;
402 
403  std::string header = ss_header.str();
404 
405  // write magic
406  write_magic(out, version[0], version[1]);
407 
408  // write header length
409  if (version[0] == 1 && version[1] == 0)
410  {
411  char header_len_le16[2];
412  uint16_t header_len = uint16_t(header.length());
413 
414  header_len_le16[0] = char((header_len >> 0) & 0xff);
415  header_len_le16[1] = char((header_len >> 8) & 0xff);
416  out.write(reinterpret_cast<char*>(header_len_le16), 2);
417  }
418  else
419  {
420  char header_len_le32[4];
421  uint32_t header_len = uint32_t(header.length());
422 
423  header_len_le32[0] = char((header_len >> 0) & 0xff);
424  header_len_le32[1] = char((header_len >> 8) & 0xff);
425  header_len_le32[2] = char((header_len >> 16) & 0xff);
426  header_len_le32[3] = char((header_len >> 24) & 0xff);
427  out.write(reinterpret_cast<char*>(header_len_le32), 4);
428  }
429 
430  out << header;
431  }

434  {
435  // read header length and convert from little endian
436  char header_len_le16[2];
437  istream.read(header_len_le16, 2);
438 
439  uint16_t header_length = uint16_t(header_len_le16[0] << 0) | uint16_t(header_len_le16[1] << 8);
440 
441  if ((magic_string_length + 2 + 2 + header_length) % 16 != 0)
442  {
443  // TODO: display warning
444  }
445 
446  std::unique_ptr<char[]> buf(new char[header_length]);
447  istream.read(buf.get(), header_length);
448  std::string header(buf.get(), header_length);
449 
450  return header;

527  template <class T, layout_type L>
528  auto cast_impl(bool check_type)
529  {
530  if (m_buffer == nullptr)
531  {
532  XTENSOR_THROW(std::runtime_error, "This npy_file has already been cast.");
533  }
534  T* ptr = reinterpret_cast<T*>(&m_buffer[0]);
535  std::vector<std::size_t> strides(m_shape.size());
536  std::size_t sz = compute_size(m_shape);
537 
538  // check if the typestring matches the given one
539  if (check_type && m_typestring != detail::build_typestring<T>())
540  {
541  XTENSOR_THROW(std::runtime_error,
542  "Cast error: formats not matching "s + m_typestring +
543  " vs "s + detail::build_typestring<T>());
544  }
545 
546  if ((L == layout_type::column_major && !m_fortran_order) ||
547  (L == layout_type::row_major && m_fortran_order))
548  {
549  XTENSOR_THROW(std::runtime_error, "Cast error: layout mismatch between npy file and requested layout.");
550  }
551 
552  compute_strides(m_shape,
553  m_fortran_order ? layout_type::column_major : layout_type::row_major,
554  strides);
555  std::vector<std::size_t> shape(m_shape);
556 
557  return std::make_tuple(ptr, sz, std::move(shape), std::move(strides));
558  }

560  template <class T, layout_type L = layout_type::dynamic>

603  inline npy_file load_npy_file(std::istream& stream)
604  {
605  // check magic bytes an version number
606  unsigned char v_major, v_minor;
607  detail::read_magic(stream, &v_major, &v_minor);
608 
609  std::string header;
610 
611  if (v_major == 1 && v_minor == 0)
612  {
613  header = detail::read_header_1_0(stream);
614  }
615  else if (v_major == 2 && v_minor == 0)
616  {
617  header = detail::read_header_2_0(stream);
618  }
619  else
620  {
621  XTENSOR_THROW(std::runtime_error, "unsupported file format version");
622  }
623 
624  // parse header
625  bool fortran_order;
626  std::string typestr;
627 
628  std::vector<std::size_t> shape;
629  detail::parse_header(header, typestr, &fortran_order, shape);
630 
631  npy_file result(shape, fortran_order, typestr);
632  // read the data
633  stream.read(result.ptr(), std::streamsize((result.n_bytes())));
634  return result;
635  }

637  template <class O, class E>
638  inline void dump_npy_stream(O& stream, const xexpression<E>& e)
639  {
640  using value_type = typename E::value_type;
641  const E& ex = e.derived_cast();
642  auto&& eval_ex = eval(ex);
643  bool fortran_order = false;
644  if (eval_ex.layout() == layout_type::column_major && eval_ex.dimension() > 1)
645  {
646  fortran_order = true;
647  }
648 
649  std::string typestring = detail::build_typestring<value_type>();
650 
651  auto shape = eval_ex.shape();
652  detail::write_header(stream, typestring, fortran_order, shape);
653 
654  std::size_t size = compute_size(shape);
655  stream.write(reinterpret_cast<const char*>(eval_ex.data()),
656  std::streamsize((sizeof(value_type) * size)));
657  }
658  } // namespace detail
