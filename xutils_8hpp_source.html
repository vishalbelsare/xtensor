367  template <class E, class C>
368  inline std::enable_if_t<!xtl::is_integral<std::decay_t<C>>::value &&
369  xtl::is_signed<typename std::decay_t<C>::value_type>::value,
370  rebind_container_t<std::size_t, std::decay_t<C>>>
371  normalize_axis(E& expr, C&& axes)
372  {
373  rebind_container_t<std::size_t, std::decay_t<C>> res;
374  resize_container(res, axes.size());
375 
376  for (std::size_t i = 0; i < axes.size(); ++i)
377  {
378  res[i] = normalize_axis(expr.dimension(), axes[i]);
379  }
380 
381  XTENSOR_ASSERT(std::all_of(res.begin(), res.end(), [&expr](auto ax_el) { return ax_el < expr.dimension(); }));
382 
383  return res;
384  }
385 
386  template <class C, class E>
387  inline std::enable_if_t<!xtl::is_integral<std::decay_t<C>>::value && std::is_unsigned<typename std::decay_t<C>::value_type>::value, C&&>
388  normalize_axis(E& expr, C&& axes)
389  {
390  static_cast<void>(expr);
391  XTENSOR_ASSERT(std::all_of(axes.begin(), axes.end(), [&expr](auto ax_el) { return ax_el < expr.dimension(); }));
392  return std::forward<C>(axes);
393  }

395  template <class R, class E, class C>
396  inline auto forward_normalize(E& expr, C&& axes)
397  -> std::enable_if_t<xtl::is_signed<std::decay_t<decltype(*std::begin(axes))>>::value, R>
398  {
399  R res;
400  xt::resize_container(res, xtl::sequence_size(axes));
401  auto dim = expr.dimension();
402  std::transform(std::begin(axes), std::end(axes), std::begin(res), [&dim](auto ax_el) {
403  return normalize_axis(dim, ax_el);
404  });
405 
406  XTENSOR_ASSERT(std::all_of(res.begin(), res.end(), [&expr](auto ax_el) { return ax_el < expr.dimension(); }));
407 
408  return res;
409  }
410 
411  template <class R, class E, class C>
412  inline auto forward_normalize(E& expr, C&& axes)
413  -> std::enable_if_t<!xtl::is_signed<std::decay_t<decltype(*std::begin(axes))>>::value && !std::is_same<R, std::decay_t<C>>::value, R>
414  {
415  static_cast<void>(expr);
416 
417  R res;
418  xt::resize_container(res, xtl::sequence_size(axes));
419  std::copy(std::begin(axes), std::end(axes), std::begin(res));
420  XTENSOR_ASSERT(std::all_of(res.begin(), res.end(), [&expr](auto ax_el) { return ax_el < expr.dimension(); }));
421  return res;
422  }
423 
424  template <class R, class E, class C>
425  inline auto forward_normalize(E& expr, C&& axes)
426  -> std::enable_if_t<!xtl::is_signed<std::decay_t<decltype(*std::begin(axes))>>::value && std::is_same<R, std::decay_t<C>>::value, R&&>
427  {
428  static_cast<void>(expr);
429  XTENSOR_ASSERT(std::all_of(std::begin(axes), std::end(axes), [&expr](auto ax_el) { return ax_el < expr.dimension(); }));
430  return std::move(axes);
431  }
