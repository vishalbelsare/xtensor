166  // if row major the view can only be (statically) computed as contiguous if:
167  // any number of integers is followed by either one or no range which
168  // are followed by explicit (or implicit) all's
169  //
170  // e.g.
171  // (i, j, all(), all()) == contiguous
172  // (i, range(0, 2), all()) == contiguous
173  // (i) == contiguous (implicit all slices)
174  // (i, all(), j) == *not* contiguous
175  // (i, range(0, 2), range(0, 2)) == *not* contiguous etc.
176  template <bool valid, bool all_seen, bool range_seen, class V>
177  struct is_contiguous_view_impl<layout_type::row_major, valid, all_seen, range_seen, V>
178  {
179  using slice = xtl::mpl::front_t<V>;
180  static constexpr bool is_range_slice = is_xrange<slice>::value;
181  static constexpr bool is_int_slice = xtl::is_integral<slice>::value;
182  static constexpr bool is_all_slice = is_xall_slice<slice>::value;
183  static constexpr bool have_all_seen = all_seen || is_all_slice;
184  static constexpr bool have_range_seen = is_range_slice;
185 
186  static constexpr bool is_valid = valid && (have_all_seen ? is_all_slice : (!range_seen && (is_int_slice || is_range_slice)));
187 
188  static constexpr bool value = is_contiguous_view_impl<layout_type::row_major,
189  is_valid,
190  have_all_seen,
191  range_seen || is_range_slice,
192  xtl::mpl::pop_front_t<V>>::value;
193  };

1643  {

1653 
1654  auto s = static_cast<diff_type>((std::min)(static_cast<size_type>(std::distance(first, last)), this->dimension()));

1659 
1660  // need to advance captured `first`

1662  std::advance(first, static_cast<diff_type>(k - xt::integral_count_before<S...>(i)));

1665  {
1666  index[i] = k < sizeof...(S) ?
1667  apply<size_type>(k, func1, m_slices) : static_cast<ivalue_type>(*first);

1671  index[i] = k < sizeof...(S) ?
